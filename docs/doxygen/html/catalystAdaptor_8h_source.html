10void catalystInitialize(domain_ *domain)
11{
12 // read input parameter
13 io_ *io = domain->io;
14 mesh_ *mesh = domain->mesh;
15
16 PetscPrintf(mesh->MESH_COMM, "Initializing TOSCA's catalyst-paraview interface...");
17
18 readDictWord ("sampling/catalystProperties", "ioType", &(io->ioTypeCatalyst));
19 readDictWord ("sampling/catalystProperties", "outputType", &(io->outputTypeCatalyst));
20 readDictDouble("sampling/catalystProperties", "startTime", &(io->startTimeCatalyst));
21 readDictDouble("sampling/catalystProperties", "timeInterval",&(io->timeIntervalCatalyst));
22
23 if(io->outputTypeCatalyst!="adjustableTime" && io->outputTypeCatalyst!="timeStep")
24 {
25 char error[512];
26 sprintf(error, "unknown catalyst outputType. Available types are\n - adjustableTime\n - timeStep\n");
27 fatalErrorInFunction("catalystInitialize", error);
28 }
29
30 // create catalyst folder
31 PetscMPIInt rank;
32 MPI_Comm_rank(PETSC_COMM_WORLD, &rank);
33
34 if(!rank)
35 {
36 errno = 0;
37 PetscInt dirRes = mkdir("./catalyst", 0777);
38 if(dirRes != 0 && errno != EEXIST)
39 {
40 char error[512];
41 sprintf(error, "could not create catalyst directory\n");
42 fatalErrorInFunction("catalystInitialize", error);
43 }
44 }
45
46 if(io->ioTypeCatalyst == "general") {}
47 else if(io->ioTypeCatalyst == "script")
48 {
49 readDictWord("sampling/catalystProperties", "scriptName", &(io->scriptNameCatalyst));
50 }
51 else
52 {
53 char error[512];
54 sprintf(error, "unknown catalyst ioType. Available types are\n - general\n - script\n");
55 fatalErrorInFunction("catalystInitialize", error);
56 }
57
58 // initialize catalyst
59 conduit_node* catalyst_init_params = conduit_node_create();
60 if(io->ioTypeCatalyst == "general")
61 {
62 std::string path2save = "catalyst/dataset-%04t.vtpd";
63 conduit_node_set_path_char8_str(catalyst_init_params, "catalyst/pipelines/0/type", "io");
64 conduit_node_set_path_char8_str(catalyst_init_params, "catalyst/pipelines/0/filename", path2save.c_str());
65 conduit_node_set_path_char8_str(catalyst_init_params, "catalyst/pipelines/0/channel", "grid");
66 }
67 else if(io->ioTypeCatalyst == "script")
68 {
69 // pass scripts on command line
70 word scriptName = "./sampling/" + io->scriptNameCatalyst;
71 conduit_node_set_path_char8_str(catalyst_init_params, "catalyst/scripts/script0", scriptName.c_str());
72 }
73
74 conduit_node_set_path_char8_str(catalyst_init_params, "catalyst_load/implementation", "paraview");
75 conduit_node_set_path_char8_str(catalyst_init_params, "catalyst_load/search_paths/paraview", PARAVIEW_IMPL_DIR);
76 enum catalyst_status err = catalyst_initialize(catalyst_init_params);
77 conduit_node_destroy(catalyst_init_params);
78
79 if (err != catalyst_status_ok)
80 {
81 char error[512];
82 sprintf(error, "\nTOSCA's catalyst-paraview interface initialization failed\n");
83 fatalErrorInFunction("catalystInitialize", error);
84 }
85
86 PetscPrintf(mesh->MESH_COMM, "done\n\n");
87}
